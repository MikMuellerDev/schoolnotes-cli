% Maths and symbols
\usepackage{amsmath}
\usepackage{siunitx}

% Plotting graphs
\usepackage{pgfplots}
\pgfplotsset{compat=1.17}


% Custom commands
\newcommand{\addpoint}[4][black]{
    \addplot [
        color=#1,
        mark=#2,
        only marks,
    ]
    coordinates {(#3)}
    node[pin=#4:{$(#3)$}]{}
    ;
}

% \desmos[axis-options]{axis-environment}
\newcommand{\plot}[2][]{
    \begin{tikzpicture}
        \begin{axis}[
            grid=both,
            grid style=solid,
            axis equal,
            scale only axis,
            minor tick num=1,
            axis lines=middle,
            xlabel={x},
            ylabel={y},
            #1
        ]
            #2
        \end{axis}
    \end{tikzpicture}
}
% \addgraph[plot-options]{formula}
\newcommand{\addgraph}[2][]{
    \addplot[color=red, #1]
    {#2}
    ;
}
% \addsamples[plot-options]{coordinates}
\newcommand{\addsamples}[2][]{
    \addplot[color=blue, mark=*, #1]
    coordinates{#2}
    ;
}
